from flask import Flask, request

app = Flask(__name__)

@app.route('/webhook', methods=['POST'])
def webhook():
    data = request.json
    print("📩 트레이딩뷰에서 받은 메시지:", data)

    # 여기에 바이낸스 주문 코드를 붙이면 됩니다.
    return {'status': 'received'}, 200

if __name__ == '__main__':
    app.run()
